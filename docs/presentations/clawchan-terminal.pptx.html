store.ts 
import { create } from 'zustand'
import { persist } from 'zustand/middleware'

interface DashboardState {
  // Intelligence Data
  markers: MapMarker[]
  aircraft: AircraftState[]
  satellites: SatellitePos[]
  
  // Layer Visibility
  layers: {
    aircraft: boolean
    satellites: boolean
    earthquakes: boolean
    disasters: boolean
  }
  
  // Actions
  setAircraft: (data: AircraftState[]) => void
  toggleLayer: (layer: string) => void
}

export const useDashboardStore = create<DashboardState>(
  persist((set) => ({
    aircraft: [],
    satellites: [],
    layers: {
      aircraft: false,
      satellites: false,
      earthquakes: true,
    },
    setAircraft: (data) => set({ aircraft: data }),
    toggleLayer: (layer) => set((state) => ({
      layers: { ...state.layers, [layer]: !state.layers[layer] }
    })),
  }), {
    name: 'clawchan-storage'
  })
)
